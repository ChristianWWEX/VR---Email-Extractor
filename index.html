  <script>

    function extractEmails() {

      const input = document.getElementById('input').value;


      // === BLACKLIST: Add emails here to exclude ===

      const blacklist = [

        'routing@wwex.com',

        // Add more like 'spam@example.com', 'noreply@company.com'

      ];


      // Extract emails via regex

      const emails = input.match(/[a-zA-Z0-9._%+-]+@[a-zA-Z0-9.-]+\.[a-z]{2,}/g);


      if (!emails || emails.length === 0) {

        showStatus('No emails found!', false);

        return;

      }


      // Remove blacklisted emails and duplicates

      const filtered = [...new Set(emails.filter(e => !blacklist.includes(e.toLowerCase())))];

      const cleaned = filtered.join(', ');


      // Replace textarea with cleaned output

      document.getElementById('input').value = cleaned;


      // Copy to clipboard

      navigator.clipboard.writeText(cleaned).then(() => {

        showStatus('Copied to clipboard!', true);

      });

    }


    function showStatus(message, success) {

      const status = document.getElementById('status');

      status.textContent = message;

      status.style.color = success ? '#4caf50' : '#ff5252';

      status.classList.add('show');

      setTimeout(() => status.classList.remove('show'), 2000);

    }

  </script>
